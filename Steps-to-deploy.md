# Steps to push to deployment

  - git clone

  - cd aryanmsr.github.io

  - source ~/.nvm/nvm.sh

  - nvm install node (version 16) 

  - npm install -g yarn 

  - yarn

  - npm install -g gatsby-cli

  - # npm install  or yarn install

  - git add .

  - git commit -am "changes"

  - gatsby build #causing problems just skip

  - npm run deploy
